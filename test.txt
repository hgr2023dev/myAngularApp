this.menus.forEach((menu: IMenu) => {
      let currentRoute = this.router.url;
      if (currentRoute.indexOf('report') >= 0) {
        menu.subMenus?.forEach((subMenu: IMenu) => {
          menu.isActive = true;
          if (subMenu.url.indexOf(currentRoute) >= 0) {
            subMenu.isActive = true;
            this.isMenuSelected = true;
            this.PageTitleDisplay = subMenu.name;
          } else {
            subMenu.isActive = false;
          }
        });
      } else if (menu.url.indexOf(currentRoute) >= 0) {
        menu.isActive = true;
        this.isMenuSelected = true;
        this.PageTitleDisplay = menu.name;
      } else {
        menu.isActive = false;
      }
    });
  }

  ngOnInit(): void {
    if (this.innerHeight > 0 ? this.innerHeight : screen.height) {
      this.minHeight = this.innerHeight + 60 + 'px';
      var contentBody = document.getElementsByClassName('.content-body');
    }
    this.checkIfRosterIsRunning();
  }

  setPageTitle(pageName: string) {
    this.PageTitleDisplay = pageName;
  }

  checkIfRosterIsRunning() {
    this.isRosterScheduleIsRunning = false;
    this.dashboardService.checkIfRosterIsRunning().subscribe(
      (response: boolean) => {
        this.isRosterScheduleIsRunning = response;
      },
      (error: any) => {}
    );
  }
}
